---
title: '<TrendingItems>'
description: 'Display trending product recommendations'
---

The `<TrendingItems>` component displays trending items based on analytics data.

## Import

```jsx
import { TrendingItems } from 'react-instantsearch';
```

## Props

<ParamField path="facetName" type="string">
  The facet name to get trending items for.

  ```jsx
  <TrendingItems facetName="category" facetValue="electronics" />
  ```
</ParamField>

<ParamField path="facetValue" type="string">
  The facet value to get trending items for.

  ```jsx
  <TrendingItems facetName="category" facetValue="electronics" />
  ```
</ParamField>

<ParamField path="limit" type="number">
  Maximum number of trending items to display.

  ```jsx
  <TrendingItems limit={5} />
  ```
</ParamField>

<ParamField path="threshold" type="number">
  Threshold for the trending score.

  ```jsx
  <TrendingItems threshold={0.5} />
  ```
</ParamField>

<ParamField path="queryParameters" type="object">
  Additional query parameters for the recommendation request.

  ```jsx
  <TrendingItems
    queryParameters={{
      analytics: true,
    }}
  />
  ```
</ParamField>

<ParamField path="fallbackParameters" type="object">
  Fallback parameters when not enough trending items are found.

  ```jsx
  <TrendingItems
    fallbackParameters={{
      filters: 'available:true',
    }}
  />
  ```
</ParamField>

<ParamField path="escapeHTML" type="boolean" default="true">
  Whether to escape HTML in the results.
</ParamField>

<ParamField path="transformItems" type="function">
  Transform the trending items before displaying them.

  ```jsx
  <TrendingItems
    transformItems={(items) =>
      items.map((item) => ({
        ...item,
        name: item.name.toUpperCase(),
      }))
    }
  />
  ```
</ParamField>

<ParamField path="itemComponent" type="React.Component">
  Custom component to render each trending item.

  ```jsx
  <TrendingItems
    itemComponent={({ item }) => (
      <div>
        <img src={item.image} alt={item.name} />
        <h4>{item.name}</h4>
        <p>${item.price}</p>
      </div>
    )}
  />
  ```
</ParamField>

<ParamField path="headerComponent" type="React.Component">
  Custom component to render the header.
</ParamField>

<ParamField path="emptyComponent" type="React.Component">
  Custom component to render when there are no trending items.
</ParamField>

<ParamField path="layoutComponent" type="React.Component">
  Custom component to control the layout of items.
</ParamField>

<ParamField path="classNames" type="object">
  CSS classes to customize the component styling.
</ParamField>

## Example

```jsx
import { InstantSearch, TrendingItems } from 'react-instantsearch';
import { liteClient as algoliasearch } from 'algoliasearch/lite';

const searchClient = algoliasearch(
  'YourApplicationID',
  'YourSearchOnlyAPIKey'
);

function App() {
  return (
    <InstantSearch searchClient={searchClient} indexName="products">
      <TrendingItems
        limit={8}
        itemComponent={({ item }) => (
          <div className="product-card">
            <img src={item.image} alt={item.name} />
            <h4>{item.name}</h4>
            <p>${item.price}</p>
            <button>View Details</button>
          </div>
        )}
      />
    </InstantSearch>
  );
}
```

## Trending Items by Category

```jsx
function CategoryPage({ category }) {
  return (
    <InstantSearch searchClient={searchClient} indexName="products">
      <TrendingItems
        facetName="category"
        facetValue={category}
        limit={6}
        itemComponent={({ item }) => (
          <div className="product-card">
            <img src={item.image} alt={item.name} />
            <h4>{item.name}</h4>
            <p>${item.price}</p>
          </div>
        )}
      />
    </InstantSearch>
  );
}
```

<Note>
  Trending Items requires Algolia Recommend to be enabled on your account and proper analytics data collection.
</Note>
